import sys
from collections import deque
input = sys.stdin.readline
R, C = map(int, input().split())
move = [(0, 1), (1, 0), (-1, 0), (0, -1)]
grid = []
result = [[-1 for i in range(C)] for i in range(R)]
targetR = 0
targetC = 0
for i in range(R):
    subList = list(map(int, input().split()))
    grid.append(subList)
    if(2 in subList):
        targetR = i
        targetC = subList.index(2)
for i in range(R):
    for j in range(C):
        if(grid[i][j] == 0):
            result[i][j] = 0
visited = set()
result[targetR][targetC] = 0
visited.add((targetR, targetC))
dq = deque()
dq.append((targetR, targetC, 0))
while dq:
    r, c, step = dq.popleft()
    for dr, dc in move:
        nr = dr+ r
        nc = dc + c
        if(nr < 0 or nr >= R): continue
        if(nc < 0 or nc >= C): continue
        if((nr, nc) in visited): continue
        if(grid[nr][nc] == 0): continue
        visited.add((nr, nc))
        result[nr][nc] = step+1
        dq.append((nr, nc, step+1))
for i in range(R):
    for j in range(C):
        print(result[i][j], end = " ")
    print()

import sys
from collections import deque
input = sys.stdin.readline

n, m = map(int, input().split())
directions = [(0, 1), (1, 0), (-1, 0), (0, -1)]

grid = []
result = [[-1] * m for _ in range(n)]
goal_r, goal_c = 0, 0

for i in range(n):
    row = list(map(int, input().split()))
    grid.append(row)
    for j in range(m):
        if row[j] == 0:
            result[i][j] = 0
        elif row[j] == 2:
            goal_r, goal_c = i, j

visited = {(goal_r, goal_c)}
result[goal_r][goal_c] = 0
dq = deque([(goal_r, goal_c)])

while dq:
    r, c = dq.popleft()
    for dr, dc in directions:
        nr, nc = r + dr, c + dc
        if 0 <= nr < n and 0 <= nc < m and (nr, nc) not in visited and grid[nr][nc] != 0:
            visited.add((nr, nc))
            result[nr][nc] = result[r][c] + 1
            dq.append((nr, nc))

for row in result:
    print(" ".join(map(str, row)))
